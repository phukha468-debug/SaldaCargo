-- Policy: Р В Р’В Р РҶР вҖҡРІвҖһСһР В Р’В Р РҺРІРӮСһР В Р’В Р СһРІРӮВҳР В Р’В Р РҺРІРӮВҳР В Р РӢР РҶР вҖҡРЎв„ўР В Р’В Р вҖҷР’ВөР В Р’В Р вҖҷР’В»Р В Р РӢР В Р вҖ° Р В Р’В Р В РІРӮВ Р В Р’В Р РҺРІРӮВҳР В Р’В Р СһРІРӮВҳР В Р’В Р РҺРІРӮВҳР В Р РӢР РҶР вҖҡРЎв„ў Р В Р РӢР РҶР вҖҡРЎв„ўР В Р’В Р РҺРІРӮСһР В Р’В Р вҖҷР’В»Р В Р РӢР В Р вҖ°Р В Р’В Р РҺРІРӮСңР В Р’В Р РҺРІРӮСһ Р В Р РӢР В РЎвҖңР В Р’В Р В РІРӮВ Р В Р’В Р РҺРІРӮСһР В Р’В Р РҺРІРӮВҳ Р В Р РӢР В РІРӮСҷР В Р’В Р вҖҷР’ВөР В Р’В Р РҶРІРӮСӣРІРӮвҖңР В Р РӢР В РЎвҖңР В Р РӢР РҶР вҖҡРІвҖһвҖ“
CREATE POLICY driver_view_own_trips ON trips
  FOR SELECT
  USING (driver_id = auth.uid() OR loader_id = auth.uid());

-- Policy: Р В Р’В Р РҶР вҖҡРІвҖһСһР В Р’В Р РҺРІРӮСһР В Р’В Р СһРІРӮВҳР В Р’В Р РҺРІРӮВҳР В Р РӢР РҶР вҖҡРЎв„ўР В Р’В Р вҖҷР’ВөР В Р’В Р вҖҷР’В»Р В Р РӢР В Р вҖ° Р В Р’В Р В РІРӮВ Р В Р’В Р РҺРІРӮВҳР В Р’В Р СһРІРӮВҳР В Р’В Р РҺРІРӮВҳР В Р РӢР РҶР вҖҡРЎв„ў Р В Р РӢР РҶР вҖҡРЎв„ўР В Р’В Р РҺРІРӮСһР В Р’В Р вҖҷР’В»Р В Р РӢР В Р вҖ°Р В Р’В Р РҺРІРӮСңР В Р’В Р РҺРІРӮСһ Р В Р’В Р вҖҷР’В·Р В Р’В Р вҖҷР’В°Р В Р’В Р РҺРІРӮСңР В Р’В Р вҖҷР’В°Р В Р’В Р вҖҷР’В·Р В Р РӢР РҶР вҖҡРІвҖһвҖ“ Р В Р’В Р РҺРІРӮВҳР В Р’В Р вҖҷР’В· Р В Р РӢР В РЎвҖңР В Р’В Р В РІРӮВ Р В Р’В Р РҺРІРӮСһР В Р’В Р РҺРІРӮВҳР В Р РӢР РҶР вҖҡР’ВҰ Р В Р РӢР В РІРӮСҷР В Р’В Р вҖҷР’ВөР В Р’В Р РҶРІРӮСӣРІРӮвҖңР В Р РӢР В РЎвҖңР В Р’В Р РҺРІРӮСһР В Р’В Р В РІРӮВ

  FOR SELECT
  USING (
    trip_id IN (
      SELECT id FROM trips 
      WHERE driver_id = auth.uid() OR loader_id = auth.uid()
    )
  );

-- Policy: Р В Р’В Р РҶР вҖҡРІвҖһСһР В Р’В Р РҺРІРӮСһР В Р’В Р СһРІРӮВҳР В Р’В Р РҺРІРӮВҳР В Р РӢР РҶР вҖҡРЎв„ўР В Р’В Р вҖҷР’ВөР В Р’В Р вҖҷР’В»Р В Р РӢР В Р вҖ° Р В Р’В Р РҺР’ВҳР В Р’В Р РҺРІРӮСһР В Р’В Р вҖҷР’В¶Р В Р’В Р вҖҷР’ВөР В Р РӢР РҶР вҖҡРЎв„ў Р В Р РӢР В РЎвҖңР В Р’В Р РҺРІРӮСһР В Р’В Р вҖҷР’В·Р В Р’В Р СһРІРӮВҳР В Р’В Р вҖҷР’В°Р В Р’В Р В РІРӮВ Р В Р’В Р вҖҷР’В°Р В Р РӢР РҶР вҖҡРЎв„ўР В Р РӢР В Р вҖ° Р В Р’В Р вҖҷР’В·Р В Р’В Р вҖҷР’В°Р В Р’В Р РҺРІРӮСңР В Р’В Р вҖҷР’В°Р В Р’В Р вҖҷР’В·Р В Р РӢР РҶР вҖҡРІвҖһвҖ“ Р В Р РӢР РҶР вҖҡРЎв„ўР В Р’В Р РҺРІРӮСһР В Р’В Р вҖҷР’В»Р В Р РӢР В Р вҖ°Р В Р’В Р РҺРІРӮСңР В Р’В Р РҺРІРӮСһ Р В Р’В Р В РІРӮВ  Р В Р РӢР В РЎвҖңР В Р’В Р В РІРӮВ Р В Р’В Р РҺРІРӮСһР В Р’В Р РҺРІРӮВҳР В Р РӢР РҶР вҖҡР’ВҰ Р В Р РӢР В РІРӮСҷР В Р’В Р вҖҷР’ВөР В Р’В Р РҶРІРӮСӣРІРӮвҖңР В Р РӢР В РЎвҖңР В Р’В Р вҖҷР’В°Р В Р РӢР РҶР вҖҡР’ВҰ

## 5. Р В Р’В Р РҺРІвҖһСһР В Р’В Р вҖҷР’В°Р В Р’В Р РҺРІРӮСң Р В Р РӢР В РЎвҖңР В Р РӢР РҶР вҖҡР РӢР В Р’В Р РҺРІРӮВҳР В Р РӢР РҶР вҖҡРЎв„ўР В Р’В Р вҖҷР’В°Р В Р РӢР В РІРӮв„–Р В Р РӢР РҶР вҖҡРЎв„ўР В Р РӢР В РЎвҖңР В Р РӢР В Р РҸ Р В Р’В Р вҖҷР’ВұР В Р’В Р вҖҷР’В°Р В Р’В Р вҖҷР’В»Р В Р’В Р вҖҷР’В°Р В Р’В Р В РІРӮВҰР В Р РӢР В РЎвҖңР В Р РӢР РҶР вҖҡРІвҖһвҖ“ (Р В Р’В Р РҺРІвҖһСһР В Р’В Р вҖҷР’В Р В Р’В Р вҖҷР’ВҳР В Р’В Р РҺРЎвҖәР В Р’В Р вҖҷР’ВҳР В Р’В Р вҖҷР’В§Р В Р’В Р РҺРЎСҷР В Р’В Р РҺРІРӮС”!)

```sql
-- Р В Р’В Р РҶР вҖҡР’ВҳР В Р’В Р вҖҷР’В°Р В Р’В Р вҖҷР’В»Р В Р’В Р вҖҷР’В°Р В Р’В Р В РІРӮВҰР В Р РӢР В РЎвҖң Р В Р’В Р РҺРІРӮСңР В Р’В Р РҺРІРӮСһР В Р РӢР РҶРІРӮСҷР’В¬Р В Р’В Р вҖҷР’ВөР В Р’В Р вҖҷР’В»Р В Р РӢР В Р вҖ°Р В Р’В Р РҺРІРӮСңР В Р’В Р вҖҷР’В° = SUM Р В Р’В Р В РІРӮВ Р В Р РӢР В РЎвҖңР В Р’В Р вҖҷР’ВөР В Р РӢР РҶР вҖҡР’ВҰ Р В Р РӢР РҶР вҖҡРЎв„ўР В Р РӢР В РІРӮСҷР В Р’В Р вҖҷР’В°Р В Р’В Р В РІРӮВҰР В Р’В Р вҖҷР’В·Р В Р’В Р вҖҷР’В°Р В Р’В Р РҺРІРӮСңР В Р РӢР РҶР вҖҡР’В Р В Р’В Р РҺРІРӮВҳР В Р’В Р РҶРІРӮСӣРІРӮвҖң, Р В Р’В Р РҺРІРӮвҖңР В Р’В Р СһРІРӮВҳР В Р’В Р вҖҷР’Вө Р В Р РӢР В Р РүР В Р РӢР РҶР вҖҡРЎв„ўР В Р’В Р РҺРІРӮСһ Р В Р’В Р РҺРІРӮСңР В Р’В Р РҺРІРӮСһР В Р РӢР РҶРІРӮСҷР’В¬Р В Р’В Р вҖҷР’ВөР В Р’В Р вҖҷР’В»Р В Р’В Р вҖҷР’ВөР В Р’В Р РҺРІРӮСң "to" Р В Р’В Р РҺР’ВҳР В Р’В Р РҺРІРӮВҳР В Р’В Р В РІРӮВҰР В Р РӢР РҺРІРӮСҡР В Р РӢР В РЎвҖң "from"
-- Р В Р’В Р РҺРЎвҖәР В Р’В Р РҺРІРӮС”Р В Р’В Р РҶР вҖҡРЎвҖқР В Р’В Р вҖҷР’В¬Р В Р’В Р РҺРІвҖһСһР В Р’В Р РҺРІРӮС” Р В Р’В Р РҺРІРӮвҖқР В Р’В Р РҺРІРӮСһ approved + completed

SELECT 
  wallet_id,
  SUM(
    CASE 
      WHEN to_wallet_id = wallet_id THEN amount
      WHEN from_wallet_id = wallet_id THEN -amount
      ELSE 0
    END
  ) as balance
FROM transactions
WHERE lifecycle_status = 'approved'
  AND settlement_status = 'completed'
  AND actual_date <= CURRENT_DATE
GROUP BY wallet_id;

-- P&L Р В Р’В Р вҖҷР’В·Р В Р’В Р вҖҷР’В° Р В Р’В Р РҺРІРӮвҖқР В Р’В Р вҖҷР’ВөР В Р РӢР В РІРӮСҷР В Р’В Р РҺРІРӮВҳР В Р’В Р РҺРІРӮСһР В Р’В Р СһРІРӮВҳ = Р В Р’В Р СһРІРӮВҳР В Р’В Р РҺРІРӮСһР В Р РӢР РҶР вҖҡР’ВҰР В Р’В Р РҺРІРӮСһР В Р’В Р СһРІРӮВҳР В Р РӢР РҶР вҖҡРІвҖһвҖ“ - Р В Р РӢР В РІРӮСҷР В Р’В Р вҖҷР’В°Р В Р РӢР В РЎвҖңР В Р РӢР РҶР вҖҡР’ВҰР В Р’В Р РҺРІРӮСһР В Р’В Р СһРІРӮВҳР В Р РӢР РҶР вҖҡРІвҖһвҖ“ (Р В Р РӢР РҶР вҖҡРЎв„ўР В Р’В Р РҺРІРӮСһР В Р’В Р вҖҷР’В»Р В Р РӢР В Р вҖ°Р В Р’В Р РҺРІРӮСңР В Р’В Р РҺРІРӮСһ approved + completed)
SELECT 
  business_unit_id,
  SUM(amount) FILTER (WHERE direction = 'income') as income,
  SUM(amount) FILTER (WHERE direction = 'expense') as expense,
  SUM(amount) FILTER (WHERE direction = 'income') - 
  SUM(amount) FILTER (WHERE direction = 'expense') as profit
FROM transactions
WHERE lifecycle_status = 'approved'
  AND settlement_status = 'completed'
  AND actual_date BETWEEN period_start AND period_end
GROUP BY business_unit_id;
```

CREATE OR REPLACE FUNCTION create_income_transaction()
RETURNS TRIGGER AS $$
DECLARE
  wallet_id UUID;
BEGIN
  -- Р В Р’В Р РҺРІРӮС”Р В Р’В Р РҺРІРӮвҖқР В Р РӢР В РІРӮСҷР В Р’В Р вҖҷР’ВөР В Р’В Р СһРІРӮВҳР В Р’В Р вҖҷР’ВөР В Р’В Р вҖҷР’В»Р В Р’В Р РҺРІРӮВҳР В Р РӢР РҶР вҖҡРЎв„ўР В Р РӢР В Р вҖ° Р В Р РӢР РҶР вҖҡР’В Р В Р’В Р вҖҷР’ВөР В Р’В Р вҖҷР’В»Р В Р’В Р вҖҷР’ВөР В Р’В Р В РІРӮВ Р В Р’В Р РҺРІРӮСһР В Р’В Р РҶРІРӮСӣРІРӮвҖң Р В Р’В Р РҺРІРӮСңР В Р’В Р РҺРІРӮСһР В Р РӢР РҶРІРӮСҷР’В¬Р В Р’В Р вҖҷР’ВөР В Р’В Р вҖҷР’В»Р В Р РӢР РҶР вҖҡР’ВҳР В Р’В Р РҺРІРӮСң Р В Р’В Р РҺРІРӮвҖқР В Р’В Р РҺРІРӮСһ payment_method
  wallet_id := CASE NEW.payment_method
    WHEN 'cash' THEN (SELECT id FROM wallets WHERE code = 'driver_wallet')
    WHEN 'qr' THEN (SELECT id FROM wallets WHERE code = 'ip_rs')
    WHEN 'bank_invoice' THEN (SELECT id FROM wallets WHERE code = 'ip_rs')
    WHEN 'debt_cash' THEN (SELECT id FROM wallets WHERE code = 'cash_office')
    WHEN 'card_driver' THEN (SELECT id FROM wallets WHERE code = 'driver_wallet')
  END;

  INSERT INTO transactions (
    direction, amount, from_wallet_id, to_wallet_id,
    category_id, trip_id,
    lifecycle_status, settlement_status,
    actual_date
  )
  VALUES (
    'income', NEW.amount,
    (SELECT id FROM wallets WHERE code = 'ext_clients'), -- fake external
    wallet_id,
    (SELECT id FROM categories WHERE code = 'FREIGHT_LCV_CITY' LIMIT 1),
    NEW.trip_id,
    'draft',
    CASE NEW.payment_method WHEN 'bank_invoice' THEN 'pending' 
                            WHEN 'debt_cash' THEN 'pending'
                            ELSE 'completed' END,
    CURRENT_DATE
  );

  UPDATE trip_orders 
  SET linked_income_tx_id = (SELECT id FROM transactions ORDER BY created_at DESC LIMIT 1)
  WHERE id = NEW.id;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE OR REPLACE FUNCTION check_maintenance_alert()
RETURNS TRIGGER AS $$
BEGIN
  UPDATE maintenance_alerts
  SET alert_status = CASE
    WHEN NEW.odometer_current >= next_service_mileage THEN 'overdue'
    WHEN NEW.odometer_current >= next_service_mileage - 500 THEN 'pending'
    ELSE 'pending'
  END
  WHERE asset_id = NEW.id;
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;
